## 硬件
STM32F103C8T6 

rr_wwdg.h
```c
#ifndef __RR_WWDG_H__
#define __RR_WWDG_H__

#include "stm32f103xb.h"
#include <stdint.h>

#define PCLK1_HZ        36000000UL

/* 单步计数周期（µs） */
#define WWDG_TICK_US(tb)        ((4096UL * (1UL << (tb))) * 1000000UL / PCLK1_HZ)
/* 硬件超时（µs）：计数器从 T0 跑到 0x40 */
#define WWDG_TIMEOUT_US(tb, t0) (((uint32_t)(t0) - 0x40U + 1U) * WWDG_TICK_US(tb))

/*
 * 参数化初始化
 *   win_start_ms : 窗口开始时间（喂狗最早时间），建议 > 0
 *   win_end_ms   : 窗口结束时间（喂狗最晚时间），硬件上限 58ms
 *                  若需要更长时间，设 soft_window_ms > 58
 *   soft_window_ms : 软件扩展窗口（ms），0 = 不使用软件扩展
 *                    若 > 0，表示"允许在 soft_window_ms 内任意时刻喂狗"
 *
 *  纯硬件模式（EWI 直接喂狗）
 *   └─► 只能检测：CPU 彻底死掉（死循环卡死、HardFault 跑飞）
 *   └─► 无法检测：业务逻辑卡住、任务超时、通信超时等"软性故障"
 *
 *  软件扩展模式（EWI 检查 flag）
 *   └─► 能检测：业务逻辑是否在规定时间内"打卡"
 *   └─► 本质是：证明主循环还在正常跑业务，而不只是 CPU 还活着
 */
void WWDG_Init(uint32_t win_start_ms, uint32_t win_end_ms, uint32_t soft_window_ms);

/* 主业务调用：通知"我还活着" */
void WWDG_Feed(void);

/* EWI 中断处理（内部使用，放在 wwdg.c 里） */
void WWDG_IRQHandler(void);

#endif

```

rr_wwdg.c
```c
#include "wwdg/rr_wwdg.h"
#include "debug_log.h"

/* ── 内部状态 ─────────────────────────────────── */
static uint32_t s_soft_window_ms = 0;   /* 软件窗口大小，0=不启用 */
static uint32_t s_last_feed_ms   = 0;   /* 上次业务喂狗的时间戳   */

/* 外部提供的毫秒时间戳（来自 SysTick） */
extern uint32_t get_tick_ms(void);

/* ── WWDG_Init ────────────────────────────────── */
/*
 * 参数说明（硬件层，仅影响 WWDG 本身）：
 *   win_start_ms : 窗口前沿（喂太早复位），范围 0 ~ 58ms
 *   win_end_ms   : 窗口后沿（超时复位）  ，范围 1 ~ 58ms，必须 > win_start_ms
 *   soft_window_ms : 0=纯硬件窗口；>58=软件扩展，如填 10000 即允许 10 秒内喂狗
 *
 * 使用示例：
 *   WWDG_Init(10, 50, 0);       // 10ms~50ms 内喂狗，纯硬件窗口
 *   WWDG_Init(0,  50, 10000);   // 软件扩展：10 秒内喂一次即可
 */
void WWDG_Init(uint32_t win_start_ms, uint32_t win_end_ms, uint32_t soft_window_ms)
{
    /* ① 选择最大分频 WDGTB=3，单步 ≈ 910µs */
    const uint8_t  WDGTB  = 3;
    const uint32_t tick_us = WWDG_TICK_US(WDGTB);   /* ≈ 910 µs */

    /* ② 根据 win_end_ms 反推 T0（计数器初值） */
    uint32_t end_us = win_end_ms * 1000UL;
    uint32_t steps  = end_us / tick_us;
    if (steps > 64) steps = 64;          /* 硬件上限 64 步 */
    if (steps < 1 ) steps = 1;
    uint8_t T0 = (uint8_t)(0x3FU + steps);   /* T0 = 0x40 + steps - 1，最大 0x7F */
    if (T0 > 0x7F) T0 = 0x7F;

    /* ③ 根据 win_start_ms 反推窗口寄存器 W */
    uint32_t start_us   = win_start_ms * 1000UL;
    uint32_t start_steps = start_us / tick_us;
    uint8_t W = (uint8_t)(T0 - start_steps);
    if (W < 0x40) W = 0x40;   /* W 不能低于 0x40 */
    if (W > T0  ) W = T0;

    /* ④ 保存软件窗口参数 */
    s_soft_window_ms = soft_window_ms;
    s_last_feed_ms   = get_tick_ms();

    /* ⑤ 使能 WWDG 时钟 */
    RCC->APB1ENR |= RCC_APB1ENR_WWDGEN;

    /* ⑥ 写 CFR：WDGTB + EWI + W */
    WWDG->CFR = ((uint32_t)WDGTB << 7)
              | WWDG_CFR_EWI
              | (uint32_t)W;

    /* ⑦ 写 CR：激活 + 计数器初值 */
    WWDG->CR = WWDG_CR_WDGA | (uint32_t)T0;

    /* ⑧ 使能 NVIC */
    NVIC_SetPriority(WWDG_IRQn, 0);
    NVIC_EnableIRQ(WWDG_IRQn);
}

/* ── 业务层喂狗（主循环调用） ─────────────────── */
void WWDG_Feed(void)
{
    s_last_feed_ms = get_tick_ms();
}

/* ── EWI 中断：~58ms 触发一次 ────────────────── */
void WWDG_IRQHandler(void)
{
    if (WWDG->SR & WWDG_SR_EWIF)
    {
        if (s_soft_window_ms > 0)
        {
        	/* 软件扩展模式：检查业务层是否在大窗口内喂过狗 */
			uint32_t elapsed = get_tick_ms() - s_last_feed_ms;
			if (elapsed < s_soft_window_ms) {
				/* 业务层在窗口内喂过狗 → EWI 继续代为维持硬件喂狗, 重置计时器，防止复位重启程序*/
				WWDG->CR = WWDG_CR_WDGA | 0x7FU;
			}
			/* else：超时或未喂狗 → 不喂 → 系统将在 ~58ms 后复位 */
        }
        else {
        	/* 58ms 自动喂一次*/
        	WWDG->CR = WWDG_CR_WDGA | 0x7FU;
        }

        WWDG->SR &= ~WWDG_SR_EWIF;   /* EWI中断标记，必须清标志 */
    }
}
```

main.c
```c
#include "main.h"
#include "debug_log.h"

#include "wwdg/rr_wwdg.h"

int main(void)
{
	//系统时钟配置
	SystemClock_Config();
	//滴塔时钟初始化
	SysTick_Init_ms();
	/* 初始化调试UART */
	debug_uart_init();

	/* 场景A：纯硬件窗口，10ms ~ 50ms 内必须喂狗, 系统使用了58ms的中断进行自动喂狗了 */
	WWDG_Init(10, 50, 0);

	/* 场景B：软件扩展，10 秒内喂一次即可（适合低频任务） */
//	WWDG_Init(0, 50, 10000);

	LOG_D("WWDG Starting ......");

	//GPIO 使能配置
	RCC->APB2ENR |= RCC_APB2ENR_IOPCEN;
	//清除 PC13 的 4 个配置位，设置为浮空输入 (0000)
	GPIOC->CRH &= ~(GPIO_CRH_MODE13 | GPIO_CRH_CNF13);
	GPIOC->CRH |= GPIO_CRH_MODE13_1;   // 10MHz 输出推挽

	while (1)
	{
		GPIOC->BSRR = GPIO_BSRR_BR13;   // LED 灭
		delay_ms(500);
		GPIOC->BSRR = GPIO_BSRR_BS13;   // LED 亮
		delay_ms(500);

		//模拟长任务过期，不影响场景A
		delay_ms(20000);

		WWDG_Feed();
		LOG_D("窗口期内喂狗!");
	}
}

void Error_Handler(void)
{
  __disable_irq();
  while (1)
  {
  }
}
```
