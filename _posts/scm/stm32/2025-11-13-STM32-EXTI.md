main.c
```c
/**
 * 外部中断练习
 */
#include "main.h"

void LED_Init(void);
void EXTI0_Init(void);

int main(void)
{
	//系统时钟配置
	SystemClock_Config();
	//滴塔时钟初始化
	SysTick_Init_ms();

	LED_Init();
	EXTI0_Init();

	while (1)
	{

	}
}

void LED_Init(void) {
	//GPIO 使能配置
	RCC->APB2ENR |= RCC_APB2ENR_IOPCEN;
	//清除 PC13 的 4 个配置位，设置为浮空输入 (0000)
	GPIOC->CRH &= ~(GPIO_CRH_MODE13 | GPIO_CRH_CNF13);
	GPIOC->CRH |= GPIO_CRH_MODE13_1;   // 10MHz 输出推挽
	// 初始态
	GPIOC->BSRR = GPIO_BSRR_BR13;   // 默认高电平（LED 灭，板载 LED 低电平亮）
}

/* ---------------------------------------------------------------
 * 外部中断初始化：PA0，下降沿触发
 * --------------------------------------------------------------- */
void EXTI0_Init(void) {
	/* 1. 使能 GPIOA 时钟（APB2ENR bit2）和 AFIO 时钟（APB2ENR bit0） */
	RCC->APB2ENR |= (1 << 2) | (1 << 0);

	/* 2. PA0 配置为上拉输入
	 *    CRL 控制 Pin0~Pin7，Pin0 对应 CRL[3:0]
	 *    MODE = 00（输入模式），CNF = 10（上拉/下拉输入）
	 *    ODR bit0 = 1 → 上拉 */
	GPIOA->CRL &= ~(0xF << 0); /* 清除 PA0 的 4 位配置 */
	GPIOA->CRL |= (0x8 << 0); /* MODE0=00, CNF0=10（上拉/下拉输入）*/
	GPIOA->ODR |= (1 << 0); /* ODR0=1 → 上拉 */

	/* 3. AFIO_EXTICR1：将 EXTI Line0 映射到 PA0
	 *    EXTICR1[3:0] = 0000 即为 PA，默认即是，显式写入保证正确 */
	AFIO->EXTICR[0] &= ~(0xF << 0); /* EXTI0 → PA */

	/* 4. EXTI 配置
	 *    IMR：中断屏蔽寄存器，bit0=1 开放 Line0 中断
	 *    RTSR：上升沿触发，bit0=0 不使用
	 *    FTSR：下降沿触发，bit0=1 使能 */
	EXTI->IMR |= (1 << 0); /* 开放 EXTI Line0 中断 */
	EXTI->RTSR &= ~(1 << 0); /* 禁用上升沿 */
	EXTI->FTSR |= (1 << 0); /* 使能下降沿 */

	/* 5. NVIC(Nested Vectored Interrupt Controller) 配置
	 *    EXTI0_IRQn = 6
	 *    ISER：中断使能寄存器，每位对应一个中断号
	 *    IP  ：优先级寄存器，每个中断 8bit，实际只用高 4bit
	 *           STM32F103 默认4位全为抢占优先级（NVIC_PriorityGroup_4）
	 *           这里设为优先级 0x10（抢占1，子优先级0） */
	NVIC->ISER[0] |= (1 << 6); /* 使能 EXTI0 中断（IRQ6） */
	NVIC->IP[6] = (1 << 4); /* 设置优先级：高4位=0001 → 抢占优先级1 */
}

/* ---------------------------------------------------------------
 * EXTI Line0 中断服务函数，声明在startup_stm32f10x_md.s
 * --------------------------------------------------------------- */
void EXTI0_IRQHandler(void)
{
    /* 判断 EXTI Line0 挂起标志（PR bit0） */
    if (EXTI->PR & (1 << 0))
    {
        /* 软件消抖：延时约 20ms（72MHz 主频） */
        for (volatile uint32_t i = 0; i < 1440000; i++);

        /* 再次确认按键仍处于按下状态（IDR bit0 = 0） */
        if (!(GPIOA->IDR & (1 << 0)))
        {
            /* 翻转 PC13：ODR bit13 取反 */
            GPIOC->ODR ^= (1 << 13);
        }

        /* 清除中断挂起标志：写 1 清零 */
        EXTI->PR = (1 << 0);
    }
}

void Error_Handler(void)
{
  __disable_irq();
  while (1)
  {
  }
}

void assert_failed(uint8_t *file, uint32_t line)
{

}
#endif
```

## 软件触发

main.c
```c
/**
 * 外部中断练习
 */
#include "main.h"

void LED_Init(void);
// 软件触发
void EXTI0_SW_Init(void);

int main(void)
{
	//系统时钟配置
	SystemClock_Config();
	//滴塔时钟初始化
	SysTick_Init_ms();

	LED_Init();
	EXTI0_SW_Init();

	while (1)
	{
		//隔2秒触发一次中断
		delay_ms(2000);
		/* 软件触发 EXTI Line0 */
		EXTI->SWIER |= (1 << 0);
	}
}

void LED_Init(void) {
	//GPIO 使能配置
	RCC->APB2ENR |= RCC_APB2ENR_IOPCEN;
	//清除 PC13 的 4 个配置位，设置为浮空输入 (0000)
	GPIOC->CRH &= ~(GPIO_CRH_MODE13 | GPIO_CRH_CNF13);
	GPIOC->CRH |= GPIO_CRH_MODE13_1;   // 10MHz 输出推挽
	// 初始态
	GPIOC->BSRR = GPIO_BSRR_BR13;   // 默认高电平（LED 灭，板载 LED 低电平亮）
}

void EXTI0_SW_Init(void)
{
    /* 软件触发不需要 GPIO，也不需要 AFIO 映射 */
    /* 只需配置 EXTI 本身和 NVIC */

    /* 1. 开放 Line0 中断 */
    EXTI->IMR  |= (1 << 0);

    /* 2. 触发方式：软件触发时 RTSR/FTSR 不影响 SWIER
     *    但建议至少配一个，否则硬件引脚误触时行为未定义 */
    EXTI->RTSR &= ~(1 << 0);
    EXTI->FTSR &= ~(1 << 0);

    /* 3. NVIC 使能 */
    NVIC->ISER[0] |= (1 << 6);
    NVIC->IP[6]    = (1 << 4);
}

/* ---------------------------------------------------------------
 * EXTI Line0 中断服务函数，声明在startup_stm32f10x_md.s
 * --------------------------------------------------------------- */
 * 
void EXTI0_IRQHandler(void)
{
    if (EXTI->PR & (1 << 0))
    {
        GPIOC->ODR ^= (1 << 13);    /* 翻转 LED */

        /* 注意：清除 PR 必须在清除 SWIER 之后，否则 SWIER 会重新置位 PR */
        EXTI->SWIER &= ~(1 << 0);   /* 先清 SWIER */
        EXTI->PR     =  (1 << 0);   /* 再清 PR（写1清零） */
    }
}

void Error_Handler(void)
{
  __disable_irq();
  while (1)
  {
  }
}

void assert_failed(uint8_t *file, uint32_t line)
{

}
#endif
```

## EXTI + NVIC (外部中断和优先级处理)

* 硬件连接：
 *   PA0 → KEY1（下降沿），抢占优先级 0（最高）
 *   PA1 → KEY2（下降沿），抢占优先级 1
 *   PA2 → KEY3（下降沿），抢占优先级 2（最低）
 *   PC13 → LED（低电平亮）
 *   每个按键另一端接 GND，引脚配置为内部上拉输入

 ![img](/assets/articles/ic/stm32/exti/exti_nvic.jpg)

Key1 优先级最高，Key2 中，Key3 最低优先级， 优先级高的可抢占优先级低的CPU资源进行处理事件。

__NOP() 是 No Operation，让 CPU 执行一条什么都不做的空指令，只消耗一个时钟周期。
 
 实现代码：main.c
 
 ```c
 #include "main.h"
#include "debug_log.h"

/* ---------------------------------------------------------------
 * 优先级抢占示例
 * 硬件连接：
 *   PA0 → KEY1（下降沿），抢占优先级 0（最高）
 *   PA1 → KEY2（下降沿），抢占优先级 1
 *   PA2 → KEY3（下降沿），抢占优先级 2（最低）
 *   PC13 → LED（低电平亮）
 *   每个按键另一端接 GND，引脚配置为内部上拉输入
 * --------------------------------------------------------------- */

/* 记录各 ISR 执行次序，方便串口/调试器观察 */
volatile uint32_t g_log[16];
volatile uint8_t  g_log_idx = 0;
#define TRACE_COUNT(x)  do { if(g_log_idx < 16) g_log[g_log_idx++] = (x); } while(0)

void Priority_Group_Init(void);
void LED_Init(void);
void EXTI_MultiLine_Init(void);
void ToPrintLog(void);

int main(void)
{
	/* 1. 时钟必须第一个初始化 */
	SystemClock_Config();
	/* 2. 优先级分组必须在任何中断使能之前设置
	 *    放在 SysTick_Init 之前，避免 SysTick 用了错误的分组 */
	Priority_Group_Init();
	/* 3. SysTick 在分组设置之后初始化 */
	SysTick_Init_ms();
	/* 4. 串口初始化 */
	debug_uart_init();
	/* 5. 外设初始化 */
	LED_Init();

	EXTI_MultiLine_Init();

	while (1)
	{
		/* 有新的 log 就打印 */
		if (g_log_idx > 0)
			ToPrintLog();
	}
}

void ToPrintLog(void) {
	/* 关中断后拷贝，避免打印过程中 ISR 继续写 g_log */
	__disable_irq();
	uint32_t snapshot[16];
	uint8_t count = g_log_idx;
	for (uint8_t i = 0; i < count; i++)
		snapshot[i] = g_log[i];
	g_log_idx = 0;
	__enable_irq();

	for (uint8_t i = 0; i < count; i++)
	    {
	        const char *desc;
	        switch (snapshot[i])
	        {
	            case 0x00: desc = "EXTI0 enter"; break;
	            case 0x0F: desc = "EXTI0 exit";  break;
	            case 0x10: desc = "EXTI1 enter"; break;
	            case 0x1F: desc = "EXTI1 exit";  break;
	            case 0x20: desc = "EXTI2 enter"; break;
	            case 0x2F: desc = "EXTI2 exit";  break;
	            default:   desc = "unknown";     break;
	        }
	        LOG_I("[%02d] 0x%02X  %s\r\n", i, snapshot[i], desc);
	    }
	    LOG_I("===========================\r\n");
}

void LED_Init(void)
{
	//GPIO 使能配置
	RCC->APB2ENR |= RCC_APB2ENR_IOPCEN;
	//清除 PC13 的 4 个配置位，设置为浮空输入 (0000)
	GPIOC->CRH &= ~(GPIO_CRH_MODE13 | GPIO_CRH_CNF13);
	GPIOC->CRH |= GPIO_CRH_MODE13_1;   // 10MHz 输出推挽
	GPIOC->BSRR = GPIO_BSRR_BR13;   // LED 灭
}

/* ---------------------------------------------------------------
 * 第一步：设置优先级分组
 * SCB->AIRCR[10:8] = 优先级分组字段（PRIGROUP）
 * 写入需要同时写入 VECTKEY=0x05FA，否则写操作被忽略
 *
 * PRIGROUP=5 → 高2位抢占(Group2)，低2位子优先级
 *   bit7:6 = 抢占优先级（0~3）
 *   bit5:4 = 子优先级（0~3）
 *   bit3:0 = 硬件未使用
 * --------------------------------------------------------------- */
void Priority_Group_Init(void)
{
    SCB->AIRCR = (0x05FA << 16)   /* VECTKEY，固定写法，缺少则写操作无效 */
               | (5 << 8);        /* PRIGROUP=5 → Group2（2位抢占+2位子优先级）*/
}

/* ---------------------------------------------------------------
 * 第三步：三路 EXTI 初始化
 * --------------------------------------------------------------- */
void EXTI_MultiLine_Init(void)
{
	/* 使能时钟：AFIO + GPIOA，不再需要 GPIOB */
	    RCC->APB2ENR |= (1 << 0)   /* AFIO  */
	                 |  (1 << 2);  /* GPIOA，PA0/PA1/PA2 都在这里 */

	    RCC->APB2RSTR |=  (1 << 0);
	    RCC->APB2RSTR &= ~(1 << 0);
	    __NOP(); __NOP(); __NOP(); __NOP();

	    /* ── PA0：KEY1，上拉输入 ── */
	    GPIOA->CRL &= ~(0xF << 0);
	    GPIOA->CRL |=  (0x8 << 0);
	    GPIOA->ODR |=  (1 << 0);

	    /* ── PA1：KEY2，上拉输入 ── */
	    GPIOA->CRL &= ~(0xF << 4);
	    GPIOA->CRL |=  (0x8 << 4);
	    GPIOA->ODR |=  (1 << 1);

	    /* ── PA2：KEY3，上拉输入── */
	    GPIOA->CRL &= ~(0xF << 8);
	    GPIOA->CRL |=  (0x8 << 8);
	    GPIOA->ODR |=  (1 << 2);

	    /* ── AFIO 映射，三路全部指向 PA ── */
	    AFIO->EXTICR[0] &= ~(0xF << 0);   /* EXTI0 = PA（0000）*/
	    AFIO->EXTICR[0] &= ~(0xF << 4);   /* EXTI1 = PA（0000）*/
	    AFIO->EXTICR[0] &= ~(0xF << 8);   /* EXTI2 = PA（0000，不再是PB）*/

	    /* 清残留 PR */
	    EXTI->PR = (1<<0)|(1<<1)|(1<<2);

	    /* 三路全部下降沿触发 */
	    EXTI->IMR  |=  (1<<0)|(1<<1)|(1<<2);
	    EXTI->FTSR |=  (1<<0)|(1<<1)|(1<<2);
	    EXTI->RTSR &= ~((1<<0)|(1<<1)|(1<<2));

	    /* NVIC 优先级不变 */
	    NVIC->IP[6] = (0 << 6) | (0 << 4);   /* EXTI0：抢占0 */
	    NVIC->IP[7] = (1 << 6) | (0 << 4);   /* EXTI1：抢占1 */
	    NVIC->IP[8] = (2 << 6) | (0 << 4);   /* EXTI2：抢占2 */

	    NVIC->ISER[0] |= (1<<6)|(1<<7)|(1<<8);
}

/* ================================================================
 * 中断服务函数
 *
 * 抢占演示逻辑：
 *   EXTI2（低优先级）执行中途 → 按下 KEY1/KEY2 → 立即被抢占
 *   EXTI1 执行中途 → 按下 KEY1 → 立即被抢占
 *   EXTI0 执行中途 → 按下 KEY2/KEY3 → 不会被抢占（优先级更高）
 *
 * 子优先级演示：
 *   若两个同抢占优先级的中断同时挂起，子优先级小的先执行
 *   但不能互相抢占（只有抢占优先级决定能否打断）
 * ================================================================ */

/* ── EXTI2 ISR：最低优先级，可被 EXTI0/EXTI1 抢占 ── */
void EXTI2_IRQHandler(void)
{
    if (EXTI->PR & (1 << 2))
    {
    	TRACE_COUNT(0x20);   /* 标记：进入 EXTI2 */

        /* 模拟耗时操作（此期间 EXTI0/EXTI1 可以打断） */
    	/* 72MHz 下约 5000000/72000000 ≈ 69ms，够用 */
        for (volatile uint32_t i = 0; i < 5000000; i++);

        GPIOC->ODR ^= (1 << 13);   /* 翻转 LED */
        TRACE_COUNT(0x2F);                  /* 标记：EXTI2 即将退出 */
        EXTI->PR = (1 << 2); /* 清标志放最后 */
    }
}

/* ── EXTI1 ISR：中等优先级，可被 EXTI0 抢占，可抢占 EXTI2 ── */
void EXTI1_IRQHandler(void)
{
    if (EXTI->PR & (1 << 1))
    {
    	TRACE_COUNT(0x10);   /* 标记：进入 EXTI1（可能是抢占了 EXTI2） */
    	/* 3000000 ≈ 42ms */
        for (volatile uint32_t i = 0; i < 5000000; i++);

        GPIOC->ODR ^= (1 << 13);
        TRACE_COUNT(0x1F);
        EXTI->PR = (1 << 1);
    }
}

/* ── EXTI0 ISR：最高优先级，不会被任何人抢占 ── */
void EXTI0_IRQHandler(void)
{
    if (EXTI->PR & (1 << 0))
    {
    	TRACE_COUNT(0x00);   /* 标记：进入 EXTI0（可能抢占了 EXTI1 或 EXTI2）*/
    	/*1000000 ≈ 14ms */
        for (volatile uint32_t i = 0; i < 5000000; i++);

        GPIOC->ODR ^= (1 << 13);
        TRACE_COUNT(0x0F);
        EXTI->PR = (1 << 0);
        /* 退出后，若之前被打断的低优先级 ISR 会自动恢复执行 */
    }
}

void Error_Handler(void)
{
  __disable_irq();
  while (1)
  {
  }
}
#ifdef USE_FULL_ASSERT
void assert_failed(uint8_t *file, uint32_t line)
{

}
#endif
 ```
