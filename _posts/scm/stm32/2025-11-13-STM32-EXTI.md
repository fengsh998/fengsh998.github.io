默认情况下LED灯亮，按下按键，实现灯亮灯灭操作，因为加了防抖（按下要达20ms），所以快速按会感觉不灵敏

main.c
```c
/**
 * 外部中断练习
 */
#include "main.h"

void LED_Init(void);
void EXTI0_Init(void);

int main(void)
{
	//系统时钟配置
	SystemClock_Config();
	//滴塔时钟初始化
	SysTick_Init_ms();

	LED_Init();
	EXTI0_Init();

	while (1)
	{

	}
}

void LED_Init(void) {
	//GPIO 使能配置
	RCC->APB2ENR |= RCC_APB2ENR_IOPCEN;
	//清除 PC13 的 4 个配置位，设置为浮空输入 (0000)
	GPIOC->CRH &= ~(GPIO_CRH_MODE13 | GPIO_CRH_CNF13);
	GPIOC->CRH |= GPIO_CRH_MODE13_1;   // 10MHz 输出推挽
	// 初始态
	GPIOC->BSRR = GPIO_BSRR_BR13;   // 默认高电平（LED 灭，板载 LED 低电平亮）
}

/* ---------------------------------------------------------------
 * 外部中断初始化：PA0，下降沿触发
 * --------------------------------------------------------------- */
void EXTI0_Init(void) {
	/* 1. 使能 GPIOA 时钟（APB2ENR bit2）和 AFIO 时钟（APB2ENR bit0） */
	RCC->APB2ENR |= (1 << 2) | (1 << 0);

	/* 2. PA0 配置为上拉输入
	 *    CRL 控制 Pin0~Pin7，Pin0 对应 CRL[3:0]
	 *    MODE = 00（输入模式），CNF = 10（上拉/下拉输入）
	 *    ODR bit0 = 1 → 上拉 */
	GPIOA->CRL &= ~(0xF << 0); /* 清除 PA0 的 4 位配置 */
	GPIOA->CRL |= (0x8 << 0); /* MODE0=00, CNF0=10（上拉/下拉输入）*/
	GPIOA->ODR |= (1 << 0); /* ODR0=1 → 上拉 */

	/* 3. AFIO_EXTICR1：将 EXTI Line0 映射到 PA0
	 *    EXTICR1[3:0] = 0000 即为 PA，默认即是，显式写入保证正确 */
	AFIO->EXTICR[0] &= ~(0xF << 0); /* EXTI0 → PA */

	/* 4. EXTI 配置
	 *    IMR：中断屏蔽寄存器，bit0=1 开放 Line0 中断
	 *    RTSR：上升沿触发，bit0=0 不使用
	 *    FTSR：下降沿触发，bit0=1 使能 */
	EXTI->IMR |= (1 << 0); /* 开放 EXTI Line0 中断 */
	EXTI->RTSR &= ~(1 << 0); /* 禁用上升沿 */
	EXTI->FTSR |= (1 << 0); /* 使能下降沿 */

	/* 5. NVIC(Nested Vectored Interrupt Controller) 配置
	 *    EXTI0_IRQn = 6
	 *    ISER：中断使能寄存器，每位对应一个中断号
	 *    IP  ：优先级寄存器，每个中断 8bit，实际只用高 4bit
	 *           STM32F103 默认4位全为抢占优先级（NVIC_PriorityGroup_4）
	 *           这里设为优先级 0x10（抢占1，子优先级0） */
	NVIC->ISER[0] |= (1 << 6); /* 使能 EXTI0 中断（IRQ6） */
	NVIC->IP[6] = (1 << 4); /* 设置优先级：高4位=0001 → 抢占优先级1 */
}

/* ---------------------------------------------------------------
 * EXTI Line0 中断服务函数，声明在startup_stm32f10x_md.s
 * --------------------------------------------------------------- */
void EXTI0_IRQHandler(void)
{
    /* 判断 EXTI Line0 挂起标志（PR bit0） */
    if (EXTI->PR & (1 << 0))
    {
        /* 软件消抖：延时约 20ms（72MHz 主频） */
        for (volatile uint32_t i = 0; i < 1440000; i++);

        /* 再次确认按键仍处于按下状态（IDR bit0 = 0） */
        if (!(GPIOA->IDR & (1 << 0)))
        {
            /* 翻转 PC13：ODR bit13 取反 */
            GPIOC->ODR ^= (1 << 13);
        }

        /* 清除中断挂起标志：写 1 清零 */
        EXTI->PR = (1 << 0);
    }
}

void Error_Handler(void)
{
  __disable_irq();
  while (1)
  {
  }
}

}
#endif
```

## 软件触发

main.c
```c
/**
 * 外部中断练习
 */
#include "main.h"

void LED_Init(void);
// 软件触发
void EXTI0_SW_Init(void);

int main(void)
{
	//系统时钟配置
	SystemClock_Config();
	//滴塔时钟初始化
	SysTick_Init_ms();

	LED_Init();
	EXTI0_SW_Init();

	while (1)
	{
		//隔2秒触发一次中断
		delay_ms(2000);
		/* 软件触发 EXTI Line0 */
		EXTI->SWIER |= (1 << 0);
	}
}

void LED_Init(void) {
	//GPIO 使能配置
	RCC->APB2ENR |= RCC_APB2ENR_IOPCEN;
	//清除 PC13 的 4 个配置位，设置为浮空输入 (0000)
	GPIOC->CRH &= ~(GPIO_CRH_MODE13 | GPIO_CRH_CNF13);
	GPIOC->CRH |= GPIO_CRH_MODE13_1;   // 10MHz 输出推挽
	// 初始态
	GPIOC->BSRR = GPIO_BSRR_BR13;   // 默认高电平（LED 灭，板载 LED 低电平亮）
}

void EXTI0_SW_Init(void)
{
    /* 软件触发不需要 GPIO，也不需要 AFIO 映射 */
    /* 只需配置 EXTI 本身和 NVIC */

    /* 1. 开放 Line0 中断 */
    EXTI->IMR  |= (1 << 0);

    /* 2. 触发方式：软件触发时 RTSR/FTSR 不影响 SWIER
     *    但建议至少配一个，否则硬件引脚误触时行为未定义 */
    EXTI->RTSR &= ~(1 << 0);
    EXTI->FTSR &= ~(1 << 0);

    /* 3. NVIC 使能 */
    NVIC->ISER[0] |= (1 << 6);
    NVIC->IP[6]    = (1 << 4);
}

/* ---------------------------------------------------------------
 * EXTI Line0 中断服务函数，声明在startup_stm32f10x_md.s
 * --------------------------------------------------------------- */
 * 
void EXTI0_IRQHandler(void)
{
    if (EXTI->PR & (1 << 0))
    {
        GPIOC->ODR ^= (1 << 13);    /* 翻转 LED */

        /* 注意：清除 PR 必须在清除 SWIER 之后，否则 SWIER 会重新置位 PR */
        EXTI->SWIER &= ~(1 << 0);   /* 先清 SWIER */
        EXTI->PR     =  (1 << 0);   /* 再清 PR（写1清零） */
    }
}

void Error_Handler(void)
{
  __disable_irq();
  while (1)
  {
  }
}

void assert_failed(uint8_t *file, uint32_t line)
{

}
#endif
```
