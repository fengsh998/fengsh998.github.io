## 硬件

STM32F103C8T6开发板 

示例通过时间钟中断来闪灯板载LED灯(PC13)

main.c代码

```c
#include "main.h"
#include "debug_log.h"

void LED_GPIO_Init(void);
void tim2_init(void);

static volatile uint32_t g_tim2_irq_cnt = 0;
static volatile uint32_t g_tim2_irq_cnt_last = 0; /* 用于主循环防重复打印 */

int main(void)
{
	//系统时钟配置
	SystemClock_Config();
	//滴塔时钟初始化
	SysTick_Init_ms();
	/* 初始化调试UART */
	debug_uart_init();
	/* 初始化 GPIO */
	LED_GPIO_Init();
	/* 启动定时器 */
	tim2_init();

	while (1)
	{
		uint32_t cnt = g_tim2_irq_cnt;
		if (cnt != g_tim2_irq_cnt_last && cnt % 5 == 0) {
			g_tim2_irq_cnt_last = cnt;
			LOG_D("中断次数 %2d", cnt);
		}
	}
}

void LED_GPIO_Init(void) {
	//GPIO 使能配置
	RCC->APB2ENR |= RCC_APB2ENR_IOPCEN;
	//清除 PC13 的 4 个配置位，设置为浮空输入 (0000)
	GPIOC->CRH &= ~(GPIO_CRH_MODE13 | GPIO_CRH_CNF13);
	GPIOC->CRH |= GPIO_CRH_MODE13_1;   // 10MHz 输出推挽
	GPIOC->BSRR = GPIO_BSRR_BR13;      // LED 灭
}

/* =========================================================
 * TIM2 定时中断初始化
 *
 * 目标：每 500ms 触发一次更新中断（UEV）
 * f_tim2  = 72MHz（APB1预分频≠1时，TIM时钟 = APB1×2 = 72MHz）
 * PSC     = 7199  → f_cnt = 72MHz/(7199+1) = 10kHz
 * ARR     = 4999  → 周期  = (4999+1)/10kHz = 500ms
 *
 * 寄存器操作顺序（重要）：
 *   先写PSC/ARR，再触发UG更新，最后清标志、使能中断、启动计数
 * ========================================================= */
void tim2_init(void) {
	/* 1. 开启 TIM2 时钟（APB1ENR bit0） */
	RCC->APB1ENR |= (1UL << 0);
	RCC->APB1ENR; /* 读一次防止总线延迟 */

	/* 2. 确保计数器停止（CEN=0），设置向上计数方向 */
	TIM2->CR1 = 0; /* 清空CR1，CEN=0, DIR=0(向上) */

	/* 3. 设置预分频值 PSC=7199
	 *    注意：PSC 有影子寄存器，需要 UEV 事件后才真正加载
	 *    范围：0~65535（16位） */
	TIM2->PSC = 7199;

	/* 4. 设置自动重载值 ARR=4999
	 *    CNT 从 0 计到 ARR 后归零并产生 UEV */
	TIM2->ARR = 4999;

	/* 5. 手动产生更新事件（UG=1），强制 PSC/ARR 立即加载到影子寄存器
	 *    同时会产生一次 UIF 标志，需要在使能中断前清除 */
	TIM2->EGR = (1UL << 0); /* UG=1，触发软件更新事件 */

	/* 6. 清除由 UG 产生的更新中断标志（UIF，SR bit0）
	 *    不清除则一启动计数器就会立即进入中断 */
	TIM2->SR = ~(1UL << 0); /* 清UIF */

	/* 7. 使能 TIM2 更新中断（DIER bit0 = UIE） */
	TIM2->DIER |= (1UL << 0); /* UIE=1 */

	/* 8. 配置 NVIC：使能 TIM2 全局中断
	 *    TIM2 IRQ号 = 28，位于 NVIC_ISER0 的 bit28
	 *    先清挂起标志，再使能，防止历史中断残留 */
	NVIC->ICPR[0] = (1UL << 28); /* 清挂起 */
	NVIC->ISER[0] = (1UL << 28); /* 使能 TIM2 中断 */

	/* 9. 启动计数器（CR1 bit0 CEN=1） */
	TIM2->CR1 |= (1UL << 0); /* CEN=1，开始计数        */
}

/* =========================================================
 * TIM2 全局中断服务函数
 *
 * 向量表入口名由启动文件（startup_stm32f103xb.s）定义，
 * 函数名必须与其完全一致，编译器通过弱符号覆盖默认处理。
 *
 * 关键步骤：
 *   1. 先判断是否是 UIF（更新中断标志），避免响应其他TIM2中断源
 *   2. 立即清除 UIF（写0），否则中断退出后会重复触发
 *   3. 执行用户逻辑
 * ========================================================= */
void TIM2_IRQHandler(void)
{
	/* 判断是否为更新中断（UIF=SR[0]） */
	if (TIM2->SR & (1UL << 0)) {

		/* 清除更新中断标志（rc_w0类型：写0清零，写1无效）
		 * 必须在退出前清除，否则立即重入中断 */
		TIM2->SR = ~(1UL << 0);

		/* 用户逻辑：翻转 PC13（板载LED）
		 * 方式1：ODR 异或翻转（非原子，有读-改-写风险，但单任务下安全）
		 * 方式2：使用 BSRR（原子操作，推荐）
		 *   高16位写1 → 对应Pin清零（BRR，Reset）
		 *   低16位写1 → 对应Pin置1（BS，Set）
		 */
		if (GPIOC->ODR & (1UL << 13)) {
			/* 当前高电平 → 拉低（点亮LED）*/
			GPIOC->BSRR = (1UL << 29); /* BRR[13]=1, Reset PC13 */
		} else {
			/* 当前低电平 → 拉高（熄灭LED）*/
			GPIOC->BSRR = (1UL << 13); /* BS[13]=1, Set PC13   */
		}

		/* 中断计数（调试用，可通过调试器查看此变量） */
		g_tim2_irq_cnt++;
	}
}

void Error_Handler(void)
{
  __disable_irq();
  while (1)
  {
  }
}
#ifdef USE_FULL_ASSERT
void assert_failed(uint8_t *file, uint32_t line)
{

}
#endif

```

整个过程分四个阶段
* ① 主程序正常运行
CPU 一直在执行 while(1) 里的代码，TIM2 的计数器在硬件上自己安静地跑，完全不占 CPU。
* ② 中断触发（UIF 拉高）
当 TIM2 的 CNT 计到 ARR（4999），硬件自动置位 UIF 标志并向 NVIC 发出中断请求。CPU 在当前指令执行完后立刻响应——不是立刻，而是当前这条指令跑完才跳。
* ③ 硬件自动保存现场（压栈）
这一步你不用写代码，Cortex-M3 硬件自动把 8 个寄存器（PC、LR、PSR、R0-R3、R12）压入栈，然后跳到 TIM2_IRQHandler。整个过程约 12 个时钟周期。
* ④ ISR 执行完毕，自动恢复现场
函数 return 时，硬件从栈里把那 8 个寄存器全部恢复，PC 跳回到主程序被打断的那条指令的下一条，主程序完全不知道自己被打断过。

### 思路：
沿着信号流反推——信号从 HSE → PLL → APB1 → TIM2 计数器 → UIF → NVIC → CPU → ISR

![图片描述](/assets/articles/ic/stm32/tim_irq/tim2_interrupt_design_thinking.svg){:width="600px" height="550px"}

* ① 时钟从哪来？— RCC 第6章
拿到一个外设，第一件事永远是问：它的时钟源是谁，频率是多少？
手册第6章有一张"时钟树图"（Clock tree），你会看到：

TIM2 挂在 APB1 总线上
APB1 的最高频率是 36 MHz，但有一个特殊规则：当 APB1 预分频系数 ≠ 1 时，TIM 的时钟 = APB1 × 2

所以我们设 APB1 /2（18 MHz），TIM2 实际时钟 = 36 MHz × 2 = 72 MHz。
如果不看这一章直接上手，你算出来的 PSC/ARR 全是错的，定时时间就不对。

* ② 定多长时间？— TIM 第14章
知道了时钟频率，下一步是问：怎么把 72 MHz 变成我想要的 500 ms 间隔？
手册第14章有计数器时序图和这个公式：
f_计数 = f_TIM / (PSC + 1)
定时周期 = (ARR + 1) / f_计数
代入 500 ms：
想要 10 kHz 计数频率 → PSC = 72MHz/10kHz - 1 = 7199
想要 500 ms 周期     → ARR = 10kHz × 0.5s - 1 = 4999
PSC 和 ARR 都是 +1 是因为它们从 0 开始计，手册时序图里画得很清楚。

* ③ 如何让值生效？— TIM 14.3 节（影子寄存器）
这是很多初学者踩的坑。手册 14.3 节说：

PSC 和 ARR 都有影子寄存器（shadow register），你写入的值不会立刻起效，必须等一次**更新事件（UEV）**才会从"预装载寄存器"搬到"影子寄存器"。

所以代码里必须手动触发一次 EGR.UG = 1，强制立即更新，否则第一个计数周期可能是乱的。但这同时会产生一次 UIF 标志，所以要在使能中断前清掉它——这就是代码里顺序很重要的原因。

* ④ 如何产生并清除中断？— TIM 14.4 节寄存器描述
手册 14.4 节是寄存器的位定义表，关键看三个寄存器：
DIER（中断使能寄存器）的 UIE 位（bit0）置 1 → 允许更新中断往 NVIC 发请求。
SR（状态寄存器）的 UIF 位（bit0）是中断标志，手册在这里标注了类型：rc_w0，意思是"读清/写0清零"——这就是为什么必须用 TIM2->SR = ~(1UL << 0) 而不是 &=。
EGR（事件产生寄存器）的 UG 位（bit0）写 1 强制触发更新事件，只写不读。

* ⑤ CPU 怎么知道要响应？— PM0056 Cortex-M3 编程手册
TIM 寄存器配好了，但 CPU 还不知道要响应这个中断，需要配置 NVIC。这部分不在 STM32F1 参考手册里，而在 ARM 的 Cortex-M3 编程手册 PM0056：

NVIC->ISER[0]：中断使能寄存器，写 1 到对应 bit 使能该 IRQ
NVIC->ICPR[0]：挂起清除寄存器，使能前先清掉历史残留

TIM2 的 IRQ 号 = 28，查的是 STM32F103 参考手册的第9章中断向量表，里面列出了所有外设的 IRQ 编号。
