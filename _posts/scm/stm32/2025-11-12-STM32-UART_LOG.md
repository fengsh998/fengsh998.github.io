debug_log.c
```c
#include "debug_log.h"
#include <stdio.h>
#include <stdarg.h>
#include <string.h>

/* STM32F103 USART1 寄存器基址 */
#define USART1_BASE     0x40013800
#define RCC_BASE        0x40021000
#define GPIOA_BASE      0x40010800

/* USART1 寄存器 */
#define USART1_SR       *(volatile uint32_t *)(USART1_BASE + 0x00)
#define USART1_DR       *(volatile uint32_t *)(USART1_BASE + 0x04)
#define USART1_BRR      *(volatile uint32_t *)(USART1_BASE + 0x08)
#define USART1_CR1      *(volatile uint32_t *)(USART1_BASE + 0x0C)
#define USART1_CR2      *(volatile uint32_t *)(USART1_BASE + 0x10)
#define USART1_CR3      *(volatile uint32_t *)(USART1_BASE + 0x14)

/* RCC 寄存器 */
#define RCC_APB2ENR     *(volatile uint32_t *)(RCC_BASE + 0x18)

/* GPIOA 寄存器 */
#define GPIOA_CRH       *(volatile uint32_t *)(GPIOA_BASE + 0x04)

/* USART1_SR 位定义 */
#define USART_SR_TXE    (1 << 7)
#define USART_SR_RXNE   (1 << 5)

/* USART1_CR1 位定义 */
#define USART_CR1_UE    (1 << 13)
#define USART_CR1_TE    (1 << 3)
#define USART_CR1_RE    (1 << 2)

/* RCC_APB2ENR 位定义 */
#define RCC_APB2ENR_USART1EN (1 << 14)
#define RCC_APB2ENR_GPIOA_EN (1 << 2)

#define LOG_BUFFER_SIZE 512
static char log_buffer[LOG_BUFFER_SIZE];

/**
 * @brief USART1初始化 - 寄存器方式
 * PA9: TX (复用推挽输出)
 * PA10: RX (浮空输入)
 * 波特率: 115200 bps
 * 数据位: 8
 * 停止们: 1
 * 时钟: 72 MHz
 */
void debug_uart_init(void)
{
    uint32_t temp;

    volatile int delay;

    /* 延迟等待时钟稳定 */
    for (delay = 0; delay < 10000; delay++);

    /* 1. 使能GPIOA和USART1时钟 */
    RCC_APB2ENR |= RCC_APB2ENR_GPIOA_EN;
    RCC_APB2ENR |= RCC_APB2ENR_USART1EN;

    /* 再延迟 */
    for (delay = 0; delay < 10000; delay++);

    /* 2. 配置PA9为TX - 复用推挽输出(50MHz) */
    /* GPIOA_CRH: PA9在bit[6:9] */
    temp = GPIOA_CRH;
    temp &= ~(0x0F << 4);      /* 清除PA9配置位 */
    temp |= (0x0B << 4);       /* 配置为复用推挽输出50MHz (1011) */
    GPIOA_CRH = temp;

    /* 3. 配置PA10为RX - 浮空输入 */
    /* GPIOA_CRH: PA10在bit[8:11] */
    temp = GPIOA_CRH;
    temp &= ~(0x0F << 8);      /* 清除PA10配置位 */
    temp |= (0x04 << 8);       /* 配置为浮空输入 (0100) */
    GPIOA_CRH = temp;

    /* 先禁用 USART1 */
    USART1_CR1 &= ~USART_CR1_UE;
    for (delay = 0; delay < 1000; delay++);
    /* 4. 配置USART1波特率 */
    /* 波特率 = fck / (16 * USARTDIV) */
    /* 115200 = 72MHz / (16 * USARTDIV) */
    /* USARTDIV = 72000000 / (16 * 115200) = 39.0625 */
    /* 整数部分: 39, 小数部分: 0.0625 * 16 = 1 */
    USART1_BRR = (39 << 4) | 1;  /* 或直接写 0x271 */

    /* 5. 配置USART1控制寄存器 */
    /* 8位数据位, 1个停止位, 无奇偶校验 */
    USART1_CR1 = 0;
    USART1_CR1 |= USART_CR1_TE;  /* 使能发送器TX */
    USART1_CR1 |= USART_CR1_RE;  /* 使能接收器RX */

    /* 6. 配置CR2 (停止位) */
    USART1_CR2 = 0;  /* 1个停止位 */

    /* 7. 配置CR3 */
    USART1_CR3 = 0;  /* 无硬件流控 */

    /* 8. 使能USART1 */
    USART1_CR1 |= USART_CR1_UE;

    for (delay = 0; delay < 10000; delay++);
}

/**
 * @brief UART发送单个字节
 */
void debug_putchar(char ch)
{
    /* 等待发送缓冲区为空 */
    while (!(USART1_SR & USART_SR_TXE));

    /* 将数据写入DR寄存器 */
    USART1_DR = (uint32_t)ch & 0xFF;
}

/**
 * @brief UART发送字符串
 */
void debug_puts(const char *str)
{
    while (*str)
    {
        debug_putchar(*str);
        if (*str == '\n')
            debug_putchar('\r');
        str++;
    }
}

/**
 * @brief 格式化打印（如printf）
 */
void debug_printf(const char *format, ...)
{
    va_list args;
    va_start(args, format);

    int len = vsnprintf(log_buffer, LOG_BUFFER_SIZE, format, args);
    va_end(args);

    if (len > 0) {
    	debug_puts(log_buffer);
    }
}

/**
 * @brief 带日志级别的打印
 */
void debug_log(uint8_t level, const char *format, ...)
{
    va_list args;
    va_start(args, format);

    const char *level_str[] = {"[DEBUG]", "[INFO ]", "[WARN ]", "[ERROR]"};

    debug_printf("%s ", level_str[level]);

    int len = vsnprintf(log_buffer, LOG_BUFFER_SIZE, format, args);
    va_end(args);

	if (len > 0) {
		debug_puts(log_buffer);
	}

    debug_putchar('\n');
}

/**
 * @brief 十六进制数据打印
 */
void debug_hex_dump(const uint8_t *data, uint32_t len)
{
    uint32_t i;

    for (i = 0; i < len; i++)
    {
        if (i % 16 == 0)
            debug_printf("\r\n%04X: ", i);
        debug_printf("%02X ", data[i]);
    }
    debug_printf("\r\n");
}
```

## EXTI + NVIC (外部中断+优先级示例)

 * 硬件连接：
 *   PA0 → KEY1（下降沿），抢占优先级 0（最高）
 *   PA1 → KEY2（下降沿），抢占优先级 1
 *   PA2 → KEY3（下降沿），抢占优先级 2（最低）
 *   PC13 → LED（低电平亮）
 *   每个按键另一端接 GND，引脚配置为内部上拉输入

__NOP() 是 No Operation，让 CPU 执行一条什么都不做的空指令，只消耗一个时钟周期。

代码：main.c
```c
#include "main.h"
#include "debug_log.h"

/* ---------------------------------------------------------------
 * 优先级抢占示例
 * 硬件连接：
 *   PA0 → KEY1（下降沿），抢占优先级 0（最高）
 *   PA1 → KEY2（下降沿），抢占优先级 1
 *   PA2 → KEY3（下降沿），抢占优先级 2（最低）
 *   PC13 → LED（低电平亮）
 *   每个按键另一端接 GND，引脚配置为内部上拉输入
 * --------------------------------------------------------------- */

/* 记录各 ISR 执行次序，方便串口/调试器观察 */
volatile uint32_t g_log[16];
volatile uint8_t  g_log_idx = 0;
#define TRACE_COUNT(x)  do { if(g_log_idx < 16) g_log[g_log_idx++] = (x); } while(0)

void Priority_Group_Init(void);
void LED_Init(void);
void EXTI_MultiLine_Init(void);
void ToPrintLog(void);

int main(void)
{
	/* 1. 时钟必须第一个初始化 */
	SystemClock_Config();
	/* 2. 优先级分组必须在任何中断使能之前设置
	 *    放在 SysTick_Init 之前，避免 SysTick 用了错误的分组 */
	Priority_Group_Init();
	/* 3. SysTick 在分组设置之后初始化 */
	SysTick_Init_ms();
	/* 4. 串口初始化 */
	debug_uart_init();
	/* 5. 外设初始化 */
	LED_Init();

	EXTI_MultiLine_Init();

	while (1)
	{
		/* 有新的 log 就打印 */
		if (g_log_idx > 0)
			ToPrintLog();
	}
}

void ToPrintLog(void) {
	/* 关中断后拷贝，避免打印过程中 ISR 继续写 g_log */
	__disable_irq();
	uint32_t snapshot[16];
	uint8_t count = g_log_idx;
	for (uint8_t i = 0; i < count; i++)
		snapshot[i] = g_log[i];
	g_log_idx = 0;
	__enable_irq();

	for (uint8_t i = 0; i < count; i++)
	    {
	        const char *desc;
	        switch (snapshot[i])
	        {
	            case 0x00: desc = "EXTI0 enter"; break;
	            case 0x0F: desc = "EXTI0 exit";  break;
	            case 0x10: desc = "EXTI1 enter"; break;
	            case 0x1F: desc = "EXTI1 exit";  break;
	            case 0x20: desc = "EXTI2 enter"; break;
	            case 0x2F: desc = "EXTI2 exit";  break;
	            default:   desc = "unknown";     break;
	        }
	        LOG_I("[%02d] 0x%02X  %s\r\n", i, snapshot[i], desc);
	    }
	    LOG_I("===========================\r\n");
}

void LED_Init(void)
{
	//GPIO 使能配置
	RCC->APB2ENR |= RCC_APB2ENR_IOPCEN;
	//清除 PC13 的 4 个配置位，设置为浮空输入 (0000)
	GPIOC->CRH &= ~(GPIO_CRH_MODE13 | GPIO_CRH_CNF13);
	GPIOC->CRH |= GPIO_CRH_MODE13_1;   // 10MHz 输出推挽
	GPIOC->BSRR = GPIO_BSRR_BR13;   // LED 灭
}

/* ---------------------------------------------------------------
 * 第一步：设置优先级分组
 * SCB->AIRCR[10:8] = 优先级分组字段（PRIGROUP）
 * 写入需要同时写入 VECTKEY=0x05FA，否则写操作被忽略
 *
 * PRIGROUP=5 → 高2位抢占(Group2)，低2位子优先级
 *   bit7:6 = 抢占优先级（0~3）
 *   bit5:4 = 子优先级（0~3）
 *   bit3:0 = 硬件未使用
 * --------------------------------------------------------------- */
void Priority_Group_Init(void)
{
    SCB->AIRCR = (0x05FA << 16)   /* VECTKEY，固定写法，缺少则写操作无效 */
               | (5 << 8);        /* PRIGROUP=5 → Group2（2位抢占+2位子优先级）*/
}

/* ---------------------------------------------------------------
 * 第三步：三路 EXTI 初始化
 * --------------------------------------------------------------- */
void EXTI_MultiLine_Init(void)
{
	/* 使能时钟：AFIO + GPIOA，不再需要 GPIOB */
	    RCC->APB2ENR |= (1 << 0)   /* AFIO  */
	                 |  (1 << 2);  /* GPIOA，PA0/PA1/PA2 都在这里 */

	    RCC->APB2RSTR |=  (1 << 0);
	    RCC->APB2RSTR &= ~(1 << 0);
	    __NOP(); __NOP(); __NOP(); __NOP();

	    /* ── PA0：KEY1，上拉输入 ── */
	    GPIOA->CRL &= ~(0xF << 0);
	    GPIOA->CRL |=  (0x8 << 0);
	    GPIOA->ODR |=  (1 << 0);

	    /* ── PA1：KEY2，上拉输入 ── */
	    GPIOA->CRL &= ~(0xF << 4);
	    GPIOA->CRL |=  (0x8 << 4);
	    GPIOA->ODR |=  (1 << 1);

	    /* ── PA2：KEY3，上拉输入── */
	    GPIOA->CRL &= ~(0xF << 8);
	    GPIOA->CRL |=  (0x8 << 8);
	    GPIOA->ODR |=  (1 << 2);

	    /* ── AFIO 映射，三路全部指向 PA ── */
	    AFIO->EXTICR[0] &= ~(0xF << 0);   /* EXTI0 = PA（0000）*/
	    AFIO->EXTICR[0] &= ~(0xF << 4);   /* EXTI1 = PA（0000）*/
	    AFIO->EXTICR[0] &= ~(0xF << 8);   /* EXTI2 = PA（0000，不再是PB）*/

	    /* 清残留 PR */
	    EXTI->PR = (1<<0)|(1<<1)|(1<<2);

	    /* 三路全部下降沿触发 */
	    EXTI->IMR  |=  (1<<0)|(1<<1)|(1<<2);
	    EXTI->FTSR |=  (1<<0)|(1<<1)|(1<<2);
	    EXTI->RTSR &= ~((1<<0)|(1<<1)|(1<<2));

	    /* NVIC 优先级不变 */
	    NVIC->IP[6] = (0 << 6) | (0 << 4);   /* EXTI0：抢占0 */
	    NVIC->IP[7] = (1 << 6) | (0 << 4);   /* EXTI1：抢占1 */
	    NVIC->IP[8] = (2 << 6) | (0 << 4);   /* EXTI2：抢占2 */

	    NVIC->ISER[0] |= (1<<6)|(1<<7)|(1<<8);
}

/* ================================================================
 * 中断服务函数
 *
 * 抢占演示逻辑：
 *   EXTI2（低优先级）执行中途 → 按下 KEY1/KEY2 → 立即被抢占
 *   EXTI1 执行中途 → 按下 KEY1 → 立即被抢占
 *   EXTI0 执行中途 → 按下 KEY2/KEY3 → 不会被抢占（优先级更高）
 *
 * 子优先级演示：
 *   若两个同抢占优先级的中断同时挂起，子优先级小的先执行
 *   但不能互相抢占（只有抢占优先级决定能否打断）
 * ================================================================ */

/* ── EXTI2 ISR：最低优先级，可被 EXTI0/EXTI1 抢占 ── */
void EXTI2_IRQHandler(void)
{
    if (EXTI->PR & (1 << 2))
    {
    	TRACE_COUNT(0x20);   /* 标记：进入 EXTI2 */

        /* 模拟耗时操作（此期间 EXTI0/EXTI1 可以打断） */
    	/* 72MHz 下约 5000000/72000000 ≈ 69ms，够用 */
        for (volatile uint32_t i = 0; i < 5000000; i++);

        GPIOC->ODR ^= (1 << 13);   /* 翻转 LED */
        TRACE_COUNT(0x2F);                  /* 标记：EXTI2 即将退出 */
        EXTI->PR = (1 << 2); /* 清标志放最后 */
    }
}

/* ── EXTI1 ISR：中等优先级，可被 EXTI0 抢占，可抢占 EXTI2 ── */
void EXTI1_IRQHandler(void)
{
    if (EXTI->PR & (1 << 1))
    {
    	TRACE_COUNT(0x10);   /* 标记：进入 EXTI1（可能是抢占了 EXTI2） */
    	/* 3000000 ≈ 42ms */
        for (volatile uint32_t i = 0; i < 5000000; i++);

        GPIOC->ODR ^= (1 << 13);
        TRACE_COUNT(0x1F);
        EXTI->PR = (1 << 1);
    }
}

/* ── EXTI0 ISR：最高优先级，不会被任何人抢占 ── */
void EXTI0_IRQHandler(void)
{
    if (EXTI->PR & (1 << 0))
    {
    	TRACE_COUNT(0x00);   /* 标记：进入 EXTI0（可能抢占了 EXTI1 或 EXTI2）*/
    	/*1000000 ≈ 14ms */
        for (volatile uint32_t i = 0; i < 5000000; i++);

        GPIOC->ODR ^= (1 << 13);
        TRACE_COUNT(0x0F);
        EXTI->PR = (1 << 0);
        /* 退出后，若之前被打断的低优先级 ISR 会自动恢复执行 */
    }
}

void Error_Handler(void)
{
  __disable_irq();
  while (1)
  {
  }
}
#ifdef USE_FULL_ASSERT
void assert_failed(uint8_t *file, uint32_t line)
{

}
#endif
```

EXTI0 最高优先级，EXTI1 次优先级，EXTI2 最低优先级

高优先级可以抢占低优先级执行权，多个中断到来时低优先级必须等高优先级执行完了，再来执行底优先级的。

效果：

场景1: 在按下Key0 时快速按下key1,可以看到EXIT0并没有被抢占，必须等EXTI0执行完后再执行EXIT1

```txt
[INFO ] [00] 0x00  EXTI0 enter

[INFO ] [01] 0x0F  EXTI0 exit

[INFO ] [02] 0x10  EXTI1 enter

[INFO ] [03] 0x1F  EXTI1 exit

[INFO ] ==========================
```

场景2: 在按下key1后，立即按key0,可以看到先进入EXTI1,在执行for(模拟)耗时，被EXIT0抢占了，等EXIT0执行完了再回到EXIT1继结执行。

```txt
[INFO ] [00] 0x10  EXTI1 enter

[INFO ] [01] 0x00  EXTI0 enter

[INFO ] [02] 0x0F  EXTI0 exit

[INFO ] [03] 0x1F  EXTI1 exit

[INFO ] ===========================
```
