## 硬件

STM32F103C8T6开发板 + HC-SR04 超声波测距传感器

main.c
```c
/**
 * STM32F103C8T6 — 寄存器 实现 HC-SR04 超声波测距
 *
 * 原理：
 *   1. PA1 输出 ≥10μs 高脉冲触发 HC-SR04 发射超声波
 *   2. HC-SR04 Echo 脚输出高脉冲，脉宽 = 超声波往返时间 T
 *   3. TIM2_CH1（PA0）输入捕获：
 *        上升沿 → 记录 CNT 值 t1，并切换为捕获下降沿
 *        下降沿 → 记录 CNT 值 t2
 *   4. 距离 = (t2 - t1) × f_cnt⁻¹ × 声速(340m/s) / 2
 *      f_cnt = 1MHz（PSC=71）→ 1 个计数 = 1μs
 *      距离(mm) = (t2 - t1) × 0.17   （单位μs → mm）
 *
 * 时钟：HSE 8MHz → PLL×9 → 72MHz
 * TIM2：PSC=71 → 1MHz，ARR=0xFFFF（最大量程65ms≈11m，足够）
 *
 * 串口输出（USART1，PA9=TX，115200bps）：每次测距结果打印到终端
 *
 * 硬件连接：
 *   PA0  → HC-SR04 Echo（TIM2_CH1，输入捕获）
 *   PA1  → HC-SR04 Trig（GPIO输出）
 *   PA9  → USB转TTL RX（USART1 TX，调试输出）
 *   VCC(5V) → HC-SR04 VCC
 *   GND     → HC-SR04 GND
 */

#include "main.h"
#include "debug_log.h"
#include <stdint.h>
#include <string.h>
// 在无HC-SR04硬件模拟验证
#include "sim_hcsr04.h"

/* =========================================================
 * Flash 等待周期
 * ========================================================= */
#define FLASH_LATENCY_2WS   (2UL << FLASH_ACR_LATENCY_Pos)

/* =========================================================
 * 超声波参数
 *
 * f_cnt = 1MHz → 1 count = 1μs
 * 声速 340m/s = 0.34 mm/μs
 * 往返距离 = T(μs) × 0.34 / 2 = T × 0.17 mm
 * 用整数运算：distance_mm = pulse_us × 17 / 100
 *
 * 超时判断：HC-SR04 最大量程约4m，往返≈23.5ms
 * 超过 30000μs（30ms）计数差视为超时/无反射
 * ========================================================= */
#define TRIG_PULSE_US       15U       /* Trig 脉冲宽度，≥10μs即可 */
#define ECHO_TIMEOUT_US     30000UL   /* Echo 超时阈值（30ms）      */
#define SPEED_FACTOR_NUM    17UL      /* 距离计算分子：×17          */
#define SPEED_FACTOR_DEN    100UL     /* 距离计算分母：÷100         */

/* =========================================================
 * 捕获状态机
 * ========================================================= */
typedef enum {
    CAP_IDLE    = 0,   /* 空闲，等待触发         */
    CAP_RISING  = 1,   /* 已捕获上升沿，等待下降 */
    CAP_DONE    = 2,   /* 捕获完成               */
    CAP_TIMEOUT = 3    /* 超时                   */
} CapState_t;

static volatile CapState_t  g_cap_state   = CAP_IDLE;
static volatile uint32_t    g_cap_rise    = 0;   /* 上升沿CNT值 */
static volatile uint32_t    g_cap_fall    = 0;   /* 下降沿CNT值 */
static volatile uint32_t    g_overflow_cnt = 0;  /* TIM2溢出次数 */


void gpio_trig_init(void);
void tim2_capture_init(void);
/* 脉宽可传参 */
void trig_send_pulse(uint32_t width_us);
uint32_t capture_get_pulse_us(void);

/* 0：使用模拟HC-SR04硬件，1时为真实的HC-SR04硬件 */
#define USE_SIM_MODE 0

int main(void)
{
	//系统时钟配置
	SystemClock_Config();
	//滴塔时钟初始化
	SysTick_Init_ms();

	gpio_trig_init();
	tim2_capture_init();

	/* 初始化调试UART */
	debug_uart_init();

	//GPIO 使能配置
	RCC->APB2ENR |= RCC_APB2ENR_IOPCEN;
	//清除 PC13 的 4 个配置位，设置为浮空输入 (0000)
	GPIOC->CRH   &= ~(GPIO_CRH_MODE13 | GPIO_CRH_CNF13);
	GPIOC->CRH   |= GPIO_CRH_MODE13_1;   // 10MHz 输出推挽

#if USE_SIM_MODE == 0
	uint8_t idx = 0; //模拟脉宽索引
#endif
	while (1)
	{
#if USE_SIM_MODE == 0
		uint32_t expect_us = sim_pulses_us[idx];
		trig_send_pulse(expect_us);
#else
		uint32_t =  TRIG_PULSE_US;
		/* 1. 发出 Trig 脉冲，启动一次测量 */
		trig_send_pulse(uint32_t);
#endif
		/* 2. 等待捕获完成或超时（轮询状态机） */
		uint32_t wait = 0;
		while (g_cap_state != CAP_DONE && g_cap_state != CAP_TIMEOUT) {
			delay_us(10);
			wait += 10;
			if (wait > 40000UL) { /* 40ms 保底超时 */
				g_cap_state = CAP_TIMEOUT;
				break;
			}
		}

		/* 3. 计算并输出距离 */
		if (g_cap_state == CAP_DONE) {
			uint32_t pulse_us = capture_get_pulse_us();
#if USE_SIM_MODE == 0
			int32_t err_us = (int32_t)pulse_us - (int32_t)expect_us;
			LOG_D("expect=%lu, us  captured=%lu, us err=%lu us",(unsigned long)err_us, (unsigned long)pulse_us,(unsigned long)err_us);
			/*
			 * 同步输出等效距离（按HC-SR04公式）
			 * distance_mm = captured_us × 17 / 100
			 */
			uint32_t dist_mm = pulse_us * 17UL / 100UL;
			LOG_D("(~ %lu mm)",(unsigned long)dist_mm);
#else
			if (pulse_us == 0 || pulse_us > ECHO_TIMEOUT_US) {
				LOG_D("超出测量范围");
			} else {
				uint32_t dist_mm = pulse_us * SPEED_FACTOR_NUM
						/ SPEED_FACTOR_DEN;
				LOG_D("距离为: %lu mm",(unsigned long)dist_mm);
			}
#endif
		} else {
#if USE_SIM_MODE == 0
			LOG_D("expect= %lu us TIMEOUT", expect_us);
#else
			LOG_D("测距超时!");
#endif
		}

		/* 4. 复位状态，准备下一次测量（HC-SR04 建议测量间隔≥60ms） */
		g_cap_state = CAP_IDLE;
		g_overflow_cnt = 0;
		delay_ms(100);

		// 反转 LED
		GPIOC->ODR ^= GPIO_ODR_ODR13_Msk;
	}
}

/* =========================================================
 * GPIO 初始化
 *
 * PA0：浮空输入（TIM2_CH1 输入捕获，Echo信号进来）
 *   CRL[3:0]：CNF=01（浮空输入），MODE=00（输入）→ 0x04
 *
 * PA1：推挽输出50MHz（Trig，主动发脉冲）
 *   CRL[7:4]：CNF=00（通用推挽），MODE=11（50MHz）→ 0x03
 *
 * PA9：复用推挽输出（USART1 TX）
 *   CRH[7:4]：CNF=10（复用推挽），MODE=11（50MHz）→ 0x0B
 * ========================================================= */
void gpio_trig_init(void) {
	/* 开启 GPIOA、AFIO 时钟 */
	RCC->APB2ENR |= RCC_APB2ENR_IOPAEN_Msk | RCC_APB2ENR_AFIOEN_Msk;
	(void) RCC->APB2ENR;

	/* PA0：浮空输入（CNF=01，MODE=00）*/
	GPIOA->CRL = (GPIOA->CRL & ~(GPIO_CRL_CNF0_Msk | GPIO_CRL_MODE0_Msk))
			| (1UL << GPIO_CRL_CNF0_Pos); /* CNF=01，MODE=00(默认) */

	/* PA1：通用推挽输出50MHz（CNF=00，MODE=11）*/
	GPIOA->CRL = (GPIOA->CRL & ~(GPIO_CRL_CNF1_Msk | GPIO_CRL_MODE1_Msk))
			| (3UL << GPIO_CRL_MODE1_Pos); /* CNF=00，MODE=11 */

	/* PA1 初始低电平（Trig空闲为低） */
	GPIOA->BRR = GPIO_BRR_BR1_Msk;

	/* PA9：复用推挽输出50MHz（USART1 TX）*/
	GPIOA->CRH = (GPIOA->CRH & ~(GPIO_CRH_CNF9_Msk | GPIO_CRH_MODE9_Msk))
			| (2UL << GPIO_CRH_CNF9_Pos) /* CNF=10：复用推挽 */
			| (3UL << GPIO_CRH_MODE9_Pos); /* MODE=11：50MHz   */
}

/* =========================================================
 * TIM2 输入捕获初始化
 *
 * 目标：捕获 PA0（TIM2_CH1）上 Echo 信号的上升沿和下降沿
 *
 * 参数：
 *   PSC=71 → f_cnt=1MHz（1count=1μs，与距离计算完美匹配）
 *   ARR=0xFFFF → 最大计数65535μs≈65ms（足够HC-SR04量程）
 *
 * 寄存器配置：
 *   CCMR1.CC1S=01  → CH1配置为输入，映射到TI1
 *   CCMR1.IC1F=0111 → 输入滤波，8个采样消抖（Echo信号较干净，可减小）
 *   CCER.CC1P=0    → 先捕获上升沿（极性=高电平/上升沿）
 *   CCER.CC1E=1    → 使能CH1捕获
 *   DIER.CC1IE=1   → 使能CH1捕获中断
 *   DIER.UIE=1     → 使能溢出中断（处理脉宽>65ms的超长情况）
 *
 * 中断号：
 *   TIM2_IRQn = 28（捕获+溢出共用同一个IRQ）
 * ========================================================= */
void tim2_capture_init(void)
{
    /* 开启 TIM2 时钟 */
    RCC->APB1ENR |= RCC_APB1ENR_TIM2EN_Msk;
    (void)RCC->APB1ENR;

    /* 停止计数器，清空配置 */
    TIM2->CR1  = 0;
    TIM2->DIER = 0;
    TIM2->SR   = 0;

    /* 预分频和重载值 */
    TIM2->PSC = 71U;       /* 72MHz/(71+1) = 1MHz */
    TIM2->ARR = 0xFFFFU;   /* 65535μs ≈ 65ms      */

    /*
     * CCMR1：CH1 配置为输入捕获
     *   CC1S[1:0]=01  → 输入，TI1（PA0直接映射）
     *   IC1PSC[3:2]=00 → 不分频，每个边沿都捕获
     *   IC1F[7:4]=0111 → 8个内部时钟周期采样滤波
     *
     * CMSIS宏：
     *   TIM_CCMR1_CC1S_Pos=0，TIM_CCMR1_IC1F_Pos=4
     */
    TIM2->CCMR1 = (1UL << TIM_CCMR1_CC1S_Pos)   /* CC1S=01：输入TI1 */
                | (7UL << TIM_CCMR1_IC1F_Pos);   /* IC1F=0111：8周期滤波 */

    /*
     * CCER：使能CH1，先捕获上升沿
     *   CC1E=1：捕获使能
     *   CC1P=0：上升沿触发（高电平有效）
     *   CC1NP=0（F103只有CC1P有效）
     */
    TIM2->CCER = TIM_CCER_CC1E_Msk;   /* CC1P默认0=上升沿 */

    /*
     * 使能两个中断源：
     *   CC1IE：捕获/比较1中断（上升/下降沿时触发）
     *   UIE  ：更新（溢出）中断（CNT从0xFFFF溢出归零时触发）
     */
    TIM2->DIER = TIM_DIER_CC1IE_Msk | TIM_DIER_UIE_Msk;

    /* 强制更新PSC/ARR影子寄存器，并清除随之产生的UIF */
    TIM2->EGR = TIM_EGR_UG_Msk;
    TIM2->SR  = 0;

    /* 配置NVIC：使能TIM2中断（IRQ=28，位于ISER[0] bit28） */
    NVIC->ICPR[0] = (1UL << 28U);   /* 清挂起 */
    NVIC->ISER[0] = (1UL << 28U);   /* 使能   */

    /* 启动计数器 */
    TIM2->CR1 = TIM_CR1_CEN_Msk;
}

/* =========================================================
 * 发送 Trig 脉冲（≥10μs 高脉冲）
 *
 * 同时重置捕获状态机和溢出计数，为新一轮测量做准备,HC-SR04 15μs
 * ========================================================= */
void trig_send_pulse(uint32_t width_us)
{
    /* 复位状态 */
    g_cap_state    = CAP_IDLE;
    g_overflow_cnt = 0;
    g_cap_rise     = 0;
    g_cap_fall     = 0;

    /* 清除可能残留的捕获标志 */
    TIM2->SR = 0;

    /* 切换CH1为捕获上升沿，准备好再发Trig */
    TIM2->CCER &= ~TIM_CCER_CC1P_Msk;   /* CC1P=0：上升沿 */

    /* 发出 ≥10μs 高脉冲 */
    GPIOA->BSRR = GPIO_BSRR_BS1_Msk;    /* PA1=1，Trig高 */
    delay_us(width_us > 0 ? width_us : TRIG_PULSE_US);             /* 保持15μs       */
    GPIOA->BRR  = GPIO_BRR_BR1_Msk;     /* PA1=0，Trig低  */
}

/* =========================================================
 * 计算脉冲宽度（单位：μs）
 *
 * 考虑溢出：
 *   pulse_us = (g_cap_fall - g_cap_rise) + g_overflow_cnt × 65536
 *
 * 调用前提：g_cap_state == CAP_DONE
 * ========================================================= */
uint32_t capture_get_pulse_us(void)
{
    uint32_t diff;

    if (g_cap_fall >= g_cap_rise) {
        /* 无溢出或溢出已被计数补偿（正常情况） */
        diff = g_cap_fall - g_cap_rise;
    } else {
        /* 单次溢出内发生的翻转（g_overflow_cnt已算在内） */
        diff = (0xFFFFUL - g_cap_rise) + g_cap_fall + 1UL;
    }

    /* 加上完整溢出周期（每次溢出=65536μs） */
    diff += g_overflow_cnt * 65536UL;

    return diff;
}

/* =========================================================
 * TIM2 中断服务函数（捕获 + 溢出共用）
 *
 * 处理三个事件：
 *   ① UIF（溢出）：g_overflow_cnt++，更新超时判断
 *   ② CC1IF（上升沿捕获）：记录t1，切换为下降沿捕获
 *   ③ CC1IF（下降沿捕获）：记录t2，状态→DONE，停止捕获中断
 *
 * 注意：CC1IF 在读取 CCR1 后硬件自动清除（rc_r 类型）
 *       UIF 需要软件写0清除（rc_w0 类型）
 * ========================================================= */
void TIM2_IRQHandler(void)
{
    uint32_t sr = TIM2->SR;   /* 一次性读出，避免重复判断时标志被硬件改变 */

    /* ---- ① 溢出中断 ---- */
    if (sr & TIM_SR_UIF_Msk) {
        TIM2->SR = ~TIM_SR_UIF_Msk;   /* 清UIF（写0清零） */

        if (g_cap_state == CAP_RISING) {
            g_overflow_cnt++;
            /* 溢出超过阈值，判定为超时（无回波） */
            if (g_overflow_cnt * 65536UL > ECHO_TIMEOUT_US) {
                g_cap_state = CAP_TIMEOUT;
                TIM2->DIER &= ~TIM_DIER_CC1IE_Msk;   /* 停止捕获中断 */
            }
        }
    }

    /* ---- ② 捕获中断（上升沿 或 下降沿） ---- */
    if (sr & TIM_SR_CC1IF_Msk) {
        /*
         * 读取 CCR1 会自动清除 CC1IF（rc_r：read-clear）
         * 必须先读，后判断状态，顺序不能反
         */
        uint32_t ccr_val = TIM2->CCR1;

        if (g_cap_state == CAP_IDLE) {
            /*
             * 捕获到上升沿（Echo 变高）
             * 记录时间戳，切换为下降沿捕获，更新状态
             */
            g_cap_rise  = ccr_val;
            g_overflow_cnt = 0;     /* 重置溢出计数，从上升沿开始算 */
            g_cap_state = CAP_RISING;

            /* 切换为下降沿触发：CC1P=1 */
            TIM2->CCER |= TIM_CCER_CC1P_Msk;
        }
        else if (g_cap_state == CAP_RISING) {
            /*
             * 捕获到下降沿（Echo 变低）
             * 记录时间戳，捕获完成
             */
            g_cap_fall  = ccr_val;
            g_cap_state = CAP_DONE;

            /* 停止捕获中断，避免干扰下一次测量前的空闲阶段 */
            TIM2->DIER &= ~TIM_DIER_CC1IE_Msk;
        }
    }
}

  }
}
```
