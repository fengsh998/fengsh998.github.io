## 硬件

STM32F103C8T6开发板 + LED灯 + SG90驼机


* 硬件连接：
 *   PA6  → TIM3_CH1 → RGB_R（红，串220Ω）
 *   PA7  → TIM3_CH2 → SG90(要独立5V) 信号线 / RGB_G（串220Ω）
 *   PB0  → TIM3_CH3 → RGB_B（串220Ω）
 *   PB1  → TIM3_CH4 → 扩展通道（可再接一路）
 *   PC13 → LED心跳指示

实在没有驼机的情况下只能依赖日志看ccr的值输出了。

main.c
```c
/**
 * STM32F103C8T6 — 寄存器 实现 TIM3 多路PWM输出
 *
 * 应用一：SG90 舵机（TIM3_CH2，PA7）
 *   频率：50Hz（ARR=19999，PSC=71）
 *   占空比：5%~10% → 脉宽 1ms~2ms → 角度 0°~180°
 *
 * 应用二：共阴RGB LED（TIM3_CH1/3/4，PA6/PB0/PB1）
 *   频率：1kHz（ARR=999，PSC=71）
 *   三路独立占空比控制混色
 *
 * CMSIS 使用范围（只用头文件定义，不调用库函数）：
 *   - stm32f103xb.h：外设结构体、寄存器位掩码宏（_Msk）、位位置宏（_Pos）
 *   - core_cm3.h：__NOP()、__disable_irq()、NVIC_EnableIRQ() 等内联函数
 *   所有初始化逻辑均为直接寄存器读写，不调用任何 HAL/LL 函数。
 *
 * 驼机的板载3.3V带不动。
 * 硬件连接：
 *   PA6  → TIM3_CH1 → RGB_R（红，串220Ω）
 *   PA7  → TIM3_CH2 → SG90(要独立5V) 信号线 / RGB_G（串220Ω）
 *   PB0  → TIM3_CH3 → RGB_B（串220Ω）
 *   PB1  → TIM3_CH4 → 扩展通道（可再接一路）
 *   PC13 → LED心跳指示
 */
#include "main.h"
#include "debug_log.h"

#include <stdint.h>

/* =========================================================
 * Flash 等待周期（CMSIS头文件未定义FLASH_ACR位，手动定义）
 * ========================================================= */
#define FLASH_ACR_LATENCY_2WS   (2UL << FLASH_ACR_LATENCY_Pos)
#define FLASH_ACR_PRFTBE_EN     FLASH_ACR_PRFTBE_Msk

/* =========================================================
 * PWM 参数宏
 *   舵机模式：50Hz
 *     PSC=71 → f_cnt=72MHz/72=1MHz（1μs分辨率）
 *     ARR=19999 → T=20ms=50Hz
 *     0° : CCR=1000（1ms）  90°: CCR=1500（1.5ms）  180°: CCR=2000（2ms）
 *
 *   RGB模式：1kHz
 *     PSC=71 → f_cnt=1MHz
 *     ARR=999  → T=1ms=1kHz
 *     占空比 = CCR / 1000 * 100%
 * ========================================================= */
/* 共用PSC（两种模式均为71，f_cnt=1MHz） */
#define PWM_PSC             71U

/* 舵机参数 */
#define SERVO_ARR           19999U   /* 20ms周期，50Hz */
/* 根据不同的驼机进行调整下面的值 */
#define SERVO_0DEG          600U    /* 1.0ms脉宽，0°  */
#define SERVO_180DEG        2400U    /* 2.0ms脉宽，180°*/
/* 驼机中值 */
#define SERVO_90DEG   ((SERVO_0DEG + SERVO_180DEG) / 2U)

/* RGB LED参数（ARR=999，占空比=CCR/1000） */
#define RGB_ARR             999U
#define RGB_DUTY(pct)       ((pct) * (RGB_ARR + 1) / 100U)  /* 百分比→CCR值 */

/* =========================================================
 * 运行模式选择（编译期选择，修改此宏切换两种应用）
 *   0 = SG90 舵机扫描演示
 *   1 = RGB LED 混色呼吸灯演示
 * ========================================================= */
#define APP_MODE  1   /* 0:舵机  1:RGB */

/* 初始化GPIO */
void gpio_pwm_init(void);
/* 驼机PWM配置 */
void tim3_pwm_servo_init(void);
/* 设置驼机的转动角度 */
void servo_set_angle(uint8_t angle);
/* 驼机演示*/
void servo_demo(void);
/* 调试找驼机中位值 */
void debug_find_servo_mid(void);

/* RGB混色 */
void tim3_pwm_rgb_init(void);
void rgb_set(uint8_t r, uint8_t g, uint8_t b);

int main(void)
{
	//系统时钟配置
	SystemClock_Config();
	//滴塔时钟初始化
	SysTick_Init_ms();
	/* 初始化调试UART */
	debug_uart_init();
	/* GPIO 初始化 */
	gpio_pwm_init();

#if APP_MODE == 0
	LOG_D("驼机模式!!!!");

//	debug_find_servo_mid();
	/* ---- 舵机模式,先调试找到中值后修改SERVO_0DEG，SERVO_180DEG 后再来看驼机 ---- */
	 servo_demo();
#else
	/* ---- RGB LED 模式 ---- */
	tim3_pwm_rgb_init();

	/* 预设几种颜色循环展示 */
	const uint8_t colors[][3] = { { 100, 0, 0 }, /* 红 */
	{ 0, 100, 0 }, /* 绿 */
	{ 0, 0, 100 }, /* 蓝 */
	{ 100, 100, 0 }, /* 黄 */
	{ 0, 100, 100 }, /* 青 */
	{ 100, 0, 100 }, /* 品红 */
	{ 100, 100, 100 }, /* 白 */
	};
	const uint8_t COLOR_NUM = sizeof(colors) / sizeof(colors[0]);

	while (1) {
		/* 颜色切换 */
		for (uint8_t i = 0; i < COLOR_NUM; i++) {
			rgb_set(colors[i][0], colors[i][1], colors[i][2]);
			delay_ms(1000);
		}

		/* 红色呼吸灯示例：占空比0%→100%→0% */
		for (uint16_t d = 0; d <= 100; d++) {
			rgb_set(d, 0, 0);
			delay_ms(10);
		}
		for (uint16_t d = 100; d > 0; d--) {
			rgb_set(d, 0, 0);
			delay_ms(10);
		}

		// LED翻转
		GPIOC->ODR ^= GPIO_ODR_ODR13_Msk;
	}
#endif

}

void servo_demo(void) {
	tim3_pwm_servo_init();
	servo_set_angle(90); /* 上电先回中位 */
	delay_ms(500);

	while (1) {
		/* 0° → 90° → 180° → 90° → 循环扫描 */
		servo_set_angle(0);
		delay_ms(1000);
		servo_set_angle(90);
		delay_ms(1000);
		servo_set_angle(180);
		delay_ms(1000);
		servo_set_angle(90);
		delay_ms(1000);
		// LED翻转
		GPIOC->ODR ^= GPIO_ODR_ODR13_Msk;
	}
}

/* 驼机中值调试寻找,这个需要连接上驼机，实测观察 */
void debug_find_servo_mid(void) {
	tim3_pwm_servo_init();

	uint32_t ccr = 1500;  // 从理论中位开始
	TIM3->CCR2 = ccr;
	delay_ms(500);

	while (1) {
		// 每隔1秒往上加50，观察舵机转动方向
		ccr += 50;
		if (ccr > 2500)
			ccr = 500;  // 超出范围就重置
		TIM3->CCR2 = ccr;
		LOG_D("ccr=%lu", (unsigned long )ccr);
		delay_ms(1000);
	}
}

/* =========================================================
 * GPIO 初始化 — PWM 复用推挽输出
 *
 * TIM3 通道默认引脚（无重映射）：
 *   CH1=PA6, CH2=PA7, CH3=PB0, CH4=PB1
 *
 * 复用功能输出配置（CNF[1:0]=10，通用推挽）：
 *   CRL/CRH 每4位：[CNF1:CNF0:MODE1:MODE0]
 *   复用推挽50MHz = 1011b = 0xB
 *
 * CMSIS GPIO_CRL_CNF6_Pos 等宏给出各Pin的位位置。
 * ========================================================= */
void gpio_pwm_init(void)
{
    /* 开启 GPIOA、GPIOB、GPIOC 时钟（APB2ENR） */
    RCC->APB2ENR |= RCC_APB2ENR_IOPAEN_Msk   /* GPIOA */
                  | RCC_APB2ENR_IOPBEN_Msk    /* GPIOB */
                  | RCC_APB2ENR_IOPCEN_Msk;   /* GPIOC（心跳LED）*/
    (void)RCC->APB2ENR;                        /* 读一次，消除总线延迟 */

    /*
     * PA6（TIM3_CH1）PA7（TIM3_CH2）：复用推挽输出，50MHz
     * CRL[27:24]=PA6: 0xB=1011（MODE=11,CNF=10）
     * CRL[31:28]=PA7: 0xB
     */
    GPIOA->CRL = (GPIOA->CRL
                & ~(GPIO_CRL_CNF6_Msk  | GPIO_CRL_MODE6_Msk |
                    GPIO_CRL_CNF7_Msk  | GPIO_CRL_MODE7_Msk))
               | (0x2UL << GPIO_CRL_CNF6_Pos)  | (0x3UL << GPIO_CRL_MODE6_Pos)  /* PA6:复用推挽50MHz */
               | (0x2UL << GPIO_CRL_CNF7_Pos)  | (0x3UL << GPIO_CRL_MODE7_Pos); /* PA7:复用推挽50MHz */

    /*
     * PB0（TIM3_CH3）PB1（TIM3_CH4）：复用推挽输出，50MHz
     * CRL[3:0]=PB0, CRL[7:4]=PB1
     */
    GPIOB->CRL = (GPIOB->CRL
                & ~(GPIO_CRL_CNF0_Msk  | GPIO_CRL_MODE0_Msk |
                    GPIO_CRL_CNF1_Msk  | GPIO_CRL_MODE1_Msk))
               | (0x2UL << GPIO_CRL_CNF0_Pos)  | (0x3UL << GPIO_CRL_MODE0_Pos)  /* PB0:复用推挽50MHz */
               | (0x2UL << GPIO_CRL_CNF1_Pos)  | (0x3UL << GPIO_CRL_MODE1_Pos); /* PB1:复用推挽50MHz */

    /* PC13：通用推挽输出（心跳LED），低电平点亮 */
    GPIOC->CRH = (GPIOC->CRH
                & ~(GPIO_CRH_CNF13_Msk | GPIO_CRH_MODE13_Msk))
               | (0x0UL << GPIO_CRH_CNF13_Pos) | (0x3UL << GPIO_CRH_MODE13_Pos);
    GPIOC->ODR |= GPIO_ODR_ODR13_Msk;   /* PC13初始高（熄灭） */
}

/* =========================================================
 * TIM3 PWM 初始化 — 舵机模式（50Hz，PSC=71，ARR=19999）
 *
 * 通道分工：
 *   CH2（PA7）→ SG90 信号线
 *   CH1/CH3/CH4 同样初始化为PWM，可扩展多路舵机
 *
 * 寄存器操作顺序：
 *   ① 关闭计数器（CEN=0）
 *   ② 写 PSC / ARR（影子寄存器，需UG触发加载）
 *   ③ 配置 CCMRx：OCxM=110（PWM模式1）, OCxPE=1（预装载）
 *   ④ 配置 CCER：CCxE=1（使能通道输出）
 *   ⑤ 设置初始 CCR
 *   ⑥ CR1：ARPE=1（ARR预装载）
 *   ⑦ EGR=UG 强制更新影子寄存器
 *   ⑧ CEN=1 启动
 * ========================================================= */
void tim3_pwm_servo_init(void)
{
    /* 开启 TIM3 时钟（APB1ENR bit1） */
    RCC->APB1ENR |= RCC_APB1ENR_TIM3EN_Msk;
    (void)RCC->APB1ENR;

    /* ① 停止计数器，向上计数，清空CR1 */
    TIM3->CR1 = 0;

    /* ② 预分频和自动重载 */
    TIM3->PSC = PWM_PSC;         /* 71 → f_cnt=72MHz/72=1MHz */
    TIM3->ARR = SERVO_ARR;       /* 19999 → 周期=20ms=50Hz   */

    /*
     * ③ 配置比较输出模式寄存器 CCMR1（CH1/CH2）
     *    OCxM[6:4]=110 → PWM模式1（CNT<CCR时输出有效电平）
     *    OCxPE[3]=1    → 输出比较预装载使能（CCR有影子寄存器）
     *    CCxS[1:0]=00  → 通道配置为输出方向
     *
     * CMSIS宏：TIM_CCMR1_OC1M_Pos=4, TIM_CCMR1_OC1PE_Pos=3
     *          TIM_CCMR1_OC2M_Pos=12, TIM_CCMR1_OC2PE_Pos=11
     */
    TIM3->CCMR1 = (6UL << TIM_CCMR1_OC1M_Pos)    /* CH1:PWM模式1 */
                | TIM_CCMR1_OC1PE_Msk              /* CH1:CCR预装载 */
                | (6UL << TIM_CCMR1_OC2M_Pos)      /* CH2:PWM模式1 */
                | TIM_CCMR1_OC2PE_Msk;             /* CH2:CCR预装载 */

    /* CCMR2（CH3/CH4）同样配置 */
    TIM3->CCMR2 = (6UL << TIM_CCMR2_OC3M_Pos)
                | TIM_CCMR2_OC3PE_Msk
                | (6UL << TIM_CCMR2_OC4M_Pos)
                | TIM_CCMR2_OC4PE_Msk;

    /*
     * ④ 捕获/比较使能寄存器 CCER
     *    CCxE=1：使能通道x输出
     *    CCxP=0：输出极性高（CNT<CCR时输出高）
     */
    TIM3->CCER = TIM_CCER_CC1E_Msk    /* CH1输出使能 */
               | TIM_CCER_CC2E_Msk    /* CH2输出使能 */
               | TIM_CCER_CC3E_Msk    /* CH3输出使能 */
               | TIM_CCER_CC4E_Msk;   /* CH4输出使能 */

    /* ⑤ 初始CCR：各通道输出0（高阻/低电平，避免上电乱跳） */
    TIM3->CCR1 = 0;
    TIM3->CCR2 = SERVO_90DEG;    /* CH2舵机先回中位 */
    TIM3->CCR3 = 0;
    TIM3->CCR4 = 0;

    /* ⑥ CR1：ARPE=1（ARR预装载使能），向上计数（DIR=0默认） */
    TIM3->CR1 = TIM_CR1_ARPE_Msk;

    /* ⑦ 触发UG更新，强制PSC/ARR/CCR影子寄存器立即加载 */
    TIM3->EGR = TIM_EGR_UG_Msk;

    /* ⑧ 清除UG产生的UIF标志（不需要中断） */
    TIM3->SR &= ~TIM_SR_UIF_Msk;

    /* ⑨ 启动计数器 */
    TIM3->CR1 |= TIM_CR1_CEN_Msk;
}

/* =========================================================
 * 设置舵机角度（0°~180°）
 *
 * 通道：TIM3_CH2（PA7）
 * 线性插值：CCR = SERVO_0DEG + angle*(SERVO_180DEG-SERVO_0DEG)/180
 * 写CCR时直接赋值即可（OCxPE=1，在下一个UEV时无缝更新波形）
 * ========================================================= */
void servo_set_angle(uint8_t angle)
{
    if (angle > 180U) angle = 180U;
    /* 线性映射：0°→1000us, 180°→2000us */
    uint32_t ccr = SERVO_0DEG
                 + ((uint32_t)angle * (SERVO_180DEG - SERVO_0DEG)) / 180U;
    TIM3->CCR2 = ccr;
    LOG_D("angle=%u, ccr=%lu", (unsigned)angle, (unsigned long)ccr);
}


/* =========================================================
 * TIM3 PWM 初始化 — RGB LED 模式（1kHz，PSC=71，ARR=999）
 *
 * 与舵机模式完全相同的寄存器配置流程，
 * 仅 ARR 不同（999 vs 19999），体现了 PSC 固定、ARR 决定频率。
 * ========================================================= */
void tim3_pwm_rgb_init(void)
{
    RCC->APB1ENR |= RCC_APB1ENR_TIM3EN_Msk;
    (void)RCC->APB1ENR;

    TIM3->CR1   = 0;
    TIM3->PSC   = PWM_PSC;    /* 71 → 1MHz计数频率 */
    TIM3->ARR   = RGB_ARR;    /* 999 → 1kHz PWM频率 */

    TIM3->CCMR1 = (6UL << TIM_CCMR1_OC1M_Pos) | TIM_CCMR1_OC1PE_Msk
                | (6UL << TIM_CCMR1_OC2M_Pos) | TIM_CCMR1_OC2PE_Msk;
    TIM3->CCMR2 = (6UL << TIM_CCMR2_OC3M_Pos) | TIM_CCMR2_OC3PE_Msk
                | (6UL << TIM_CCMR2_OC4M_Pos) | TIM_CCMR2_OC4PE_Msk;

    TIM3->CCER  = TIM_CCER_CC1E_Msk | TIM_CCER_CC2E_Msk
                | TIM_CCER_CC3E_Msk | TIM_CCER_CC4E_Msk;

    /* 初始全灭 */
    TIM3->CCR1 = TIM3->CCR2 = TIM3->CCR3 = TIM3->CCR4 = 0;

    TIM3->CR1  = TIM_CR1_ARPE_Msk;
    TIM3->EGR  = TIM_EGR_UG_Msk;
    TIM3->SR  &= ~TIM_SR_UIF_Msk;
    TIM3->CR1 |= TIM_CR1_CEN_Msk;
}

/* =========================================================
 * RGB LED 设置颜色（百分比，0~100）
 *
 * 通道映射（共阴RGB LED，高电平→亮）：
 *   CH1（PA6）→ R
 *   CH2（PA7）→ G
 *   CH3（PB0）→ B
 *
 * 写 CCR 立即生效（OCxPE=1保证无毛刺切换）：
 *   CCR=0   → 占空比0%  → 全灭
 *   CCR=999 → 占空比100%→ 全亮
 * ========================================================= */
void rgb_set(uint8_t r, uint8_t g, uint8_t b)
{
    TIM3->CCR1 = RGB_DUTY(r);   /* R：PA6，TIM3_CH1 */
    TIM3->CCR2 = RGB_DUTY(g);   /* G：PA7，TIM3_CH2 */
    TIM3->CCR3 = RGB_DUTY(b);   /* B：PB0，TIM3_CH3 */
}

void Error_Handler(void)
{
  __disable_irq();
  while (1)
  {
  }
}
#ifdef USE_FULL_ASSERT
void assert_failed(uint8_t *file, uint32_t line)
{

}
#endif
```
