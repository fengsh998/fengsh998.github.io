```c
#ifndef __RR_IWDG_H__
#define __RR_IWDG_H__

#include "stm32f103xb.h"   // 包含所有外设基地址和寄存器定义

#define RESET_SOURCE_NONE    0
#define RESET_SOURCE_IWDG    1
#define RESET_SOURCE_WWDG    2

/* IWDG 寄存器结构体（CMSIS 中已定义）：
   IWDG->KR  0x4000_3000
   IWDG->PR  0x4000_3004
   IWDG->RL  0x4000_3008
   IWDG->SR  0x4000_300C  */

/* 用于调试时暂停看门狗（J-Link调试必备！）*/
// DBGMCU->CR |= DBGMCU_CR_DBG_IWDG_STOP;  // 调试暂停时IWDG停止计数

/* 独立看门狗初始化
 * 使用示例：超时 = 4095 × 256/40000 ≈ 26.2s
 * IWDG_Init(6, 0xFFF);  超时26.2s
 * IWDG_Init(4, 1000);   超时 = 1000×64/40000 = 1.6s
 *
 */
void IWDG_Init(uint8_t pr, uint16_t rl);

/* 喂狗函数
 * 典型调用位置：主循环末尾、RTOS任务末尾
	while(1) {
    	App_Task1();
    	App_Task2();
    	IWDG_Feed();   // 每次循环末尾喂狗
	}
*/
void IWDG_Feed(void);

/*
 * 检测是否由IWDG复位
 * 在 main() 最开头调用，初始化外设之前
 */
uint8_t Check_Reset_Source(void);

#endif
```

rr_iwdg.c
```c
#include "iwdg/rr_iwdg.h"
#include "debug_log.h"

#define IWDG_TIMEOUT  1000U   // 循环等待上限


void IWDG_Init(uint8_t pr, uint16_t rl)
{
	uint32_t timeout;
	/* ① 写入魔法值解锁 PR/RL 寄存器 */
	IWDG->KR = 0x5555U;

	/* ② 等待解锁生效（等PVU=0）*/
	timeout = IWDG_TIMEOUT;
	while ((IWDG->SR & IWDG_SR_PVU) && (--timeout))
		;

	/* ③ 设置预分频：PR=6 → /256
	 超时 = rl × (4×2^6) / 40000 s */
	IWDG->PR = pr & 0x07U;  // 0~6，建议用宏 IWDG_PR_PR_2|PR_1|PR_0

	/* ④ 等待PR同步完成 */
	timeout = IWDG_TIMEOUT;
	while ((IWDG->SR & IWDG_SR_PVU) && (--timeout))
		;

	/* ⑤ 设置重装载值 0~4095 */
	IWDG->RLR = rl & 0x0FFFU;

	/* ⑥ 等待RLR同步完成，等待 RVU 清零 */
	timeout = IWDG_TIMEOUT;
	while ((IWDG->SR & IWDG_SR_RVU) && (--timeout))
		;

	/* ⑦ 喂狗一次（重装载计数器）*/
	IWDG->KR = 0xAAAAU;

	/* ⑧ 启动IWDG（不可逆！）*/
	IWDG->KR = 0xCCCCU;
}

void IWDG_Feed(void)
{
	/* 写入0xAAAA即重装载，无需先解锁 */
	IWDG->KR = 0xAAAAU;
	LOG_I("喂狗啦!");
}


uint8_t Check_Reset_Source(void)
{
	uint8_t source = RESET_SOURCE_NONE;

	if (RCC->CSR & RCC_CSR_IWDGRSTF) {
		/* 标记为 IWDG 复位 */
		source = RESET_SOURCE_IWDG;
		// 注意：这里先不打印，防止 UART 未就绪卡死
	}
	if (RCC->CSR & RCC_CSR_WWDGRSTF) {
		source = RESET_SOURCE_WWDG;
	}

	/* 清除 RCC 复位标志 */
	RCC->CSR |= RCC_CSR_RMVF;

	return source;
}

```

main.c
```c
/**
 * 基础框架包括：
 *    UART: 串口日志打印(占用PA9,PA10的GPIO)
 *    IWDG: 独立看门狗(Option)
 *    WWDG: 窗口看门狗(Option)
 */

#include "main.h"
#include "debug_log.h"

#include "iwdg/rr_iwdg.h"

/* 定义看门狗超时时间配置
 * 假设 LSI 时钟为 40kHz
 * 预分频 PR = 6 (即 /256)
 * 计数器频率 = 40000 / 256 = 156.25 Hz (每 6.4ms 计数一次)
 * 重装载值 RL = 156
 * 超时时间 = 156 / 156.25 ≈ 1.0 秒
 * 超时时间=(RL×预分频系数​)/LSI频率=(156×256)/40000​≈0.998s≈ 1s
 */

#define IWDG_PRESCALER    6
#define IWDG_RELOAD       156

int main(void)
{
	//系统时钟配置
	SystemClock_Config();
	//滴塔时钟初始化
	SysTick_Init_ms();
	/* 初始化调试UART */
	debug_uart_init();

	LOG_I("Start checking ...");
	// 4. 检查复位源
	//    如果上次复位是由看门狗引起的，这里会打印错误日志
	uint8_t reset_src = Check_Reset_Source();

	// 5. 初始化独立看门狗
	//    参数：预分频系数 6，重装载值 156 -> 超时时间约 1秒
	//    注意：一旦启动，不可通过软件停止
	delay_ms(5);          // 等 LSI 稳定（通常 < 2ms）
	IWDG_Init(IWDG_PRESCALER, IWDG_RELOAD);

	//GPIO 使能配置
	RCC->APB2ENR |= RCC_APB2ENR_IOPCEN;
	//清除 PC13 的 4 个配置位，设置为浮空输入 (0000)
	GPIOC->CRH   &= ~(GPIO_CRH_MODE13 | GPIO_CRH_CNF13);
	GPIOC->CRH   |= GPIO_CRH_MODE13_1;   // 10MHz 输出推挽

	LOG_I("System Running. LED Blinking...");

	// 6. 根据之前读取的状态打印复位原因
	if (reset_src == RESET_SOURCE_IWDG) {
		LOG_E("Last reset caused by IWDG!");
	} else if (reset_src == RESET_SOURCE_WWDG) {
		LOG_E("Last reset caused by WWDG!");
	}

	while (1)
	{
		GPIOC->BSRR = GPIO_BSRR_BR13;   // LED 灭
		delay_ms(500);

		// [关键] 在超时时间到达前喂狗
		// 当前配置超时约1s，此处延时500ms，非常安全
		IWDG_Feed();

		GPIOC->BSRR = GPIO_BSRR_BS13;   // LED 亮
		delay_ms(500);

		//模拟任务超时不喂狗时
		delay_ms(2000);
		// [关键] 再次喂狗，防止1sf超时
		IWDG_Feed();
	}
}

void Error_Handler(void)
{
  __disable_irq();
  while (1)
  {
  }
}

{

}
#endif
```

### 使用看门狗的场景：

软件会因为各种你预想不到的原因进入异常状态——栈溢出、野指针、电磁干扰导致 PC 指针跑飞、某个任务优先级反转死锁、外设驱动卡在等待标志位。这些问题在实验室里可能几个月都不复现，但设备数量一大、运行时间一长，概率就不可忽视。

问自己一个问题："这个设备死机了，我需要亲自去现场才能恢复它吗？" 如果答案是"是"，或者"不确定"，就必须加看门狗。设备越分散、越难抵达、越关键，看门狗就越不可缺少。
